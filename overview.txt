-- people table
CREATE TABLE IF NOT EXISTS people (
	id integer PRIMARY KEY,
	name text NOT NULL,
	height text,
	mass text,
	hair_color text,
	skin_color text,
	eye_color text,
	birth_year text,
	gender text,
	homeworld blob,
	flims blob,
	species blob,
	vehicles blob,
	created text,
	edited text,
	url text
);

-- starships table
CREATE TABLE IF NOT EXISTS starships (
	id integer PRIMARY KEY,
	name text NOT NULL,
	model text,
	manufacturer text,
	cost_in_credits text,
	length text,
	max_atmosphereing_speed text,
	crew text,
	passengers text,
	cargo_capacity text,
	consumables text,
	hypedrive_rating text,
	MGLT text,
	starship_class text,
	films blob,
	created text,
	edited text,
	url text
);

-- people_starships_relationships table
CREATE TABLE IF NOT EXISTS people_starships_relationships (
	people_reference text,
	staship_reference text
);
